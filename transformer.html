1.3 句子 → 矩阵

Satz → Matrix

一个句子的向量表示 = 将各词的嵌入向量按行堆叠成矩阵。

# 贪心最长匹配分词器（处理「垫子」等多字词）
tokenize <- function(sentence, vocab) {
  real_vocab <- sort(vocab[vocab != "<PAD>"],
                     decreasing = FALSE)          # 去掉 PAD
  # 按长度降序排列，优先匹配较长词
  real_vocab <- real_vocab[order(nchar(real_vocab), decreasing = TRUE)]
  tokens <- character(0)
  i <- 1L
  n <- nchar(sentence)
  while (i <= n) {
    matched <- FALSE
    for (v in real_vocab) {
      len <- nchar(v)
      if (i + len - 1L <= n && substr(sentence, i, i + len - 1L) == v) {
        tokens  <- c(tokens, v)
        i       <- i + len
        matched <- TRUE
        break
      }
    }
    if (!matched) i <- i + 1L  # 跳过未知字符
  }
  tokens
}

# 将词序列转为嵌入矩阵
words_to_matrix <- function(words, E) {
  idx <- word_to_idx[words] + 1L   # 0-based → 1-based
  X   <- E[idx, , drop = FALSE]
  rownames(X) <- words
  X
}

2 第二步：位置编码

Schritt 2: Positionskodierung

词嵌入有个问题：它不包含顺序信息！

  • “猫 吃 鱼” 与 “鱼 吃 猫” 的词向量集合完全相同
  • 但语义截然不同！

解决方案：给每个位置叠加一个独特的”位置信号”。

2.1 正弦位置编码

Transformer 原论文（Vaswani et al., 2017）采用三角函数：

\[ PE_{(pos,\, 2i)} = \sin\!\left(\frac{pos}{10000^{2i/d_{model}}}\right), \quad PE_{(pos,\, 2i+1)} = \cos\!\left(\frac{pos}{10000^{2i/d_{model}}}\right) \]

其中 \(pos\) 为词在句子中的位置（0-indexed），\(i\) 为维度索引（0-indexed）。

为什么选三角函数？

  • 每个位置的编码唯一
  • 相邻位置编码平滑过渡（相似度随距离单调递减）
  • 不同频率的正弦波组合，使模型能区分近距离与远距离的位置关系

3.2.2 Step 2：计算注意力分数并缩放

\[ \text{scores} = \frac{QK^\top}{\sqrt{d_k}} \]

为什么除以 \(\sqrt{d_k}\)？

\(QK^\top\) 的每个元素是 \(d_k\) 个随机变量之积的求和。若 \(Q,K\) 各元素方差为 1，则 \(QK^\top\) 元素的方差约为 \(d_k\)，标准差约为 \(\sqrt{d_k}\)。不缩放时，\(d_k\) 较大会导致 softmax 输入值极端，梯度趋近于 0（梯度消失）。

scores        <- Q %*% t(K)               # (seq_len × seq_len)
scores_scaled <- scores / sqrt(d_k)       # 缩放

3.3 封装：完整 Self-Attention 函数

self_attention <- function(X, W_Q, W_K, W_V) {
  Q <- X %*% W_Q
  K <- X %*% W_K
  V <- X %*% W_V
  d_k <- ncol(Q)

  attn_weights <- softmax_rows((Q %*% t(K)) / sqrt(d_k))
  output       <- attn_weights %*% V

  list(output = output, attention_weights = attn_weights, Q = Q, K = K, V = V)
}

输入 \(X\) \((L,\; d_{model})\)
投影 \(Q = XW_Q\) \((L,\; d_k)\)
投影 \(K = XW_K,\; V = XW_V\) \((L,\; d_k)\)
分数 \(QK^\top / \sqrt{d_k}\) \((L,\; L)\)
权重 \(\operatorname{softmax}(\cdot)\) \((L,\; L)\) — 每行和为 1
输出 \(\text{weights} \cdot V\) \((L,\; d_k)\)

\(L\) = seq_len（序列长度），最终输出形状与输入一致（当 \(d_k = d_{model}\)）。

4 第四步：多头注意力

Schritt 4: Multi-Head Attention

单个注意力头只能学习一种关注模式。语言中同时存在多种关系：语法、语义、指代……

解决方案：并行运行 \(h\) 个头，每个头各学一种模式，最后拼接融合。

\[ \text{MultiHead}(X) = \operatorname{Concat}(\text{head}_1,\ldots,\text{head}_h)\, W^O \] \[ \text{head}_i = \operatorname{Attention}(XW_Q^i,\; XW_K^i,\; XW_V^i) \]

每个头的 \(d_k = d_{model} / h\)，保证拼接后维度仍为 \(d_{model}\)。

    self_attention(X, W_Q_list[[h]], W_K_list[[h]], W_V_list[[h]])$output

  )

feed_forward <- function(X, W1, b1, W2, b2) {
  # 第一层：线性变换 + 偏置
  pre    <- sweep(X %*% W1, 2, b1, "+")   # (seq_len × d_ff)
  # ReLU：用逐元素乘法代替 pmax，保持矩阵维度不变
  hidden <- pre * (pre > 0)               # (seq_len × d_ff)
  # 第二层：线性变换 + 偏置
  sweep(hidden %*% W2, 2, b2, "+")        # (seq_len × d_model)
}

6.3 因果掩码与前向传播

Kausale Maskierung und Vorwärtsdurchlauf

训练语言模型时，位置 \(t\) 的词不能”偷看”未来（位置 \(t+1, t+2, \ldots\)）。 通过在注意力分数中加入因果掩码（将未来位置设为 \(-\infty\)）实现。

\[ \text{CausalMask}_{ij} = \begin{cases} 0 & j \le i \\ -\infty & j > i \end{cases} \]

# 因果掩码：下三角（含对角线）为 0，上三角为 -Inf
causal_mask <- function(n) {
  m <- matrix(-Inf, n, n)
  m[lower.tri(m, diag = TRUE)] <- 0
  m
}

# 带因果掩码的单头注意力
masked_self_attn <- function(X, W_Q, W_K, W_V) {
  Q <- X %*% W_Q;  K <- X %*% W_K;  V <- X %*% W_V
  d_k <- ncol(Q);  n  <- nrow(X)
  scores <- (Q %*% t(K)) / sqrt(d_k) + causal_mask(n)
  list(output = softmax_rows(scores) %*% V,
       weights = softmax_rows(scores))
}

# 前向传播
forward <- function(words, model) {
  idx <- w2i[words] + 1L
  X   <- model$E[idx, , drop = FALSE]
  L   <- nrow(X);  d <- ncol(X)

  # 词嵌入 + 位置编码
  X <- X + model$PE[seq_len(L), seq_len(d)]

  # 多头注意力（因果掩码）
  heads <- lapply(seq_along(model$W_Q), \(h)
    masked_self_attn(X, model$W_Q[[h]], model$W_K[[h]], model$W_V[[h]])$output
  )
  attn_out <- do.call(cbind, heads) %*% model$W_O
  X1 <- layer_norm(X + attn_out)

  # 前馈网络
  ffn_out <- feed_forward(X1, model$W1_ff, model$b1_ff, model$W2_ff, model$b2_ff)
  X2 <- layer_norm(X1 + ffn_out)

  # 输出层 → 概率
  logits <- sweep(X2 %*% model$W_out, 2, model$b_out, "+")
  probs  <- t(apply(logits, 1, function(r) { r <- r - max(r); e <- exp(r); e / sum(e) }))

  list(probs = probs, X2 = X2)
}

6.4 损失函数与训练循环

使用交叉熵损失：

\[ \mathcal{L} = -\frac{1}{N}\sum_{t=1}^{N} \log P(\text{目标词}_t \mid \text{前缀}_t) \]

训练使用数值梯度（有限差分逼近）演示梯度下降原理。 实际中，框架（PyTorch/TensorFlow）通过反向传播自动计算精确梯度，效率高数千倍。

# 计算损失
compute_loss <- function(model, sents) {
  total <- 0;  n <- 0L
  for (s in sents) {
    res <- forward(s, model)
    for (i in seq_len(length(s) - 1L)) {
      t_idx <- w2i[s[i + 1L]] + 1L
      total <- total - log(max(res$probs[i, t_idx], 1e-10))
      n <- n + 1L
    }
  }
  total / n
}

# 数值梯度训练
# 每轮只更新 E（词嵌入）、W_out（输出权重）、b_out（输出偏置）
# 这三组参数直接决定「什么词嵌入被映射为多高的预测概率」
train <- function(model, sents, n_epochs = 30, lr = 0.4, eps = 1e-5) {
  loss_hist <- numeric(n_epochs)

  for (ep in seq_len(n_epochs)) {
    loss_hist[ep] <- compute_loss(model, sents)
    if (ep == 1 || ep %% 5 == 0)
      cat(sprintf("Epoch %3d | Loss: %.4f\n", ep, loss_hist[ep]))

    # 更新辅助函数（数值梯度 + 梯度下降）
    update_param <- function(param) {
      for (k in seq_along(param)) {
        orig <- param[k]
        param[k] <- orig + eps;  lp <- compute_loss(model, sents)
        param[k] <- orig - eps;  lm <- compute_loss(model, sents)
        param[k] <- orig - lr * (lp - lm) / (2 * eps)
      }
      param
    }

    model$E     <- update_param(model$E)
    model$W_out <- update_param(model$W_out)
    model$b_out <- update_param(model$b_out)
  }

  list(model = model, loss = loss_hist)
}

# 1. 行级 Softmax（数值稳定）
softmax_rows <- function(X) {
  X <- X - apply(X, 1, max)
  e <- exp(X)
  e / rowSums(e)
}

# 2. 正弦位置编码
create_position_encoding <- function(max_len, d_model) {
  pos <- 0:(max_len - 1)
  PE  <- matrix(0, max_len, d_model)
  for (k in seq(1, d_model, by = 2)) {
    denom   <- 10000^((k - 1) / d_model)
    PE[, k] <- sin(pos / denom)
    if (k + 1 <= d_model) PE[, k + 1] <- cos(pos / denom)
  }
  PE
}

# 3. 层归一化（总体方差）
layer_norm <- function(X, eps = 1e-6) {
  mu    <- rowMeans(X)
  X_ctr <- sweep(X, 1, mu, "-")
  sigma <- sqrt(rowMeans(X_ctr^2) + eps)
  sweep(X_ctr, 1, sigma, "/")
}

# 4. 因果掩码（防止模型看到未来词）
causal_mask <- function(n) {
  m <- matrix(-Inf, n, n)
  m[lower.tri(m, diag = TRUE)] <- 0
  m
}

# 5. 带因果掩码的自注意力
masked_self_attn <- function(X, W_Q, W_K, W_V) {
  Q <- X %*% W_Q;  K <- X %*% W_K;  V <- X %*% W_V
  n <- nrow(X);    d_k <- ncol(Q)
  scores <- (Q %*% t(K)) / sqrt(d_k) + causal_mask(n)
  A <- softmax_rows(scores)
  list(output = A %*% V, weights = A)
}

# 6. 多头注意力
multi_head_attention <- function(X, W_Q_list, W_K_list, W_V_list, W_O) {
  heads <- lapply(seq_along(W_Q_list), \(h)
    masked_self_attn(X, W_Q_list[[h]], W_K_list[[h]], W_V_list[[h]])$output
  )
  do.call(cbind, heads) %*% W_O
}

# 7. 前馈网络
feed_forward <- function(X, W1, b1, W2, b2) {
  hidden <- pmax(0, sweep(X %*% W1, 2, b1, "+"))
  sweep(hidden %*% W2, 2, b2, "+")
}

# 8. 完整 Transformer 编码器层
transformer_layer <- function(X, W_Q_list, W_K_list, W_V_list, W_O,
                               W1_ff, b1_ff, W2_ff, b2_ff) {
  attn_out <- multi_head_attention(X, W_Q_list, W_K_list, W_V_list, W_O)
  X1 <- layer_norm(X + attn_out)
  ffn_out <- feed_forward(X1, W1_ff, b1_ff, W2_ff, b2_ff)
  layer_norm(X1 + ffn_out)
}
